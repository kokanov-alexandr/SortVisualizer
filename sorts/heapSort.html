function heapSort(array) {
    let size = array.length
    
    for (let i = Math.floor(size / 2 - 1); i >= 0; i--)
        heapify(array, size, i)
    
    for (let i = size - 1; i >= 0; i--) {
        let temp = array[0]

        array[i] = temp
        heapify(array, i, 0)
    }
}
    
function heapify(array, size, i) {
    let max = i
    let left = 2 * i + 1
    let right = 2 * i + 2
    
    if (left < size && array[left] > array[max])
        max = left
    
    if (right < size && array[right] > array[max])
        max = right
    
    if (max != i) {
        let temp = array[i]
        array[i] = array[max]
        array[max] = temp
    
        heapify(array, size, max)
    }
}            
                    
                
                
def heapify(arr, n, i):
    largest = i
    l = 2 * i + 1
    r = 2 * i + 2
    
    if l < n and arr[i] < arr[l]:
        largest = l
    
    if r < n and arr[largest] < arr[r]:
        largest = r
    
    if largest != i:
        arr[i], arr[largest] = arr[largest], arr[i]
        heapify(arr, n, largest)

void heapify(int arr[], int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if (left < n && arr[left] > arr[largest])
        largest = left;

    if (right < n && arr[right] > arr[largest])
        largest = right;

    if (largest != i) {
        swap(arr[i], arr[largest]);
        heapify(arr, n, largest);
    }
}

void heapSort(int arr[], int n) {
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    for (int i = n - 1; i >= 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}

Общее описание

Сортировка кучей, или Heap Sort, — это эффективный алгоритм сортировки, который использует структуру данных «куча» для упорядочивания элементов. Алгоритм основан на двух ключевых операциях: построении кучи (heapify) и извлечении максимального элемента из кучи (или минимального, в зависимости от типа кучи). Сортировка кучей часто используется благодаря своей устойчивой производительности и относительно простой реализации.

Основные шаги алгоритма

  • Построить максимальную кучу (max-heap) из несортированного массива. В куче родительский элемент всегда больше или равен своим потомкам.
  • Повторять процесс извлечения максимального элемента из кучи и перемещения его в конец массива, уменьшая размер кучи.
  • После каждого извлечения, восстановить структуру кучи для оставшихся элементов (процесс heapify).
  • Продолжать, пока все элементы не будут перемещены из кучи в отсортированную часть массива.

Сложность алгоритма

Сценарий Сложность
Средний и худший случай O(n * log(n))
Лучший случай O(n * log(n))

Сортировка кучей является одним из самых эффективных алгоритмов с временной сложностью O(n * log (n)) в среднем, худшем и лучшем случаях. Это делает её предпочтительным выбором для многих приложений, где требуется высокая производительность на больших массивах. Алгоритм не требует дополнительной памяти, так как выполняется на месте, и хорошо подходит для ситуаций, когда важно минимизировать использование памяти.
